def load_and_preprocess(image_path):
    try:
        img = Image.open(image_path).convert('L')
        img = img.resize((28, 28))
        img_array = np.array(img) / 255.0
        return img_array
    except:
        return None

print("Görüntüler yükleniyor...")

# Latin görüntüleri (ilk 1000 tane)
X_latin = []
y_latin = []
for _, row in latin_df.iterrows():
    img_array = load_and_preprocess(row['image_path'])
    if img_array is not None:
        X_latin.append(img_array)
        y_latin.append(row['label'])
    if len(X_latin) >= 1000:
        break

X_latin = np.array(X_latin)
y_latin = np.array(y_latin)

# Kiril görüntüleri (ilk 1000 tane)
X_cyrillic = []
y_cyrillic_text = []
for _, row in cyrillic_df.iterrows():
    img_array = load_and_preprocess(row['image_path'])
    if img_array is not None:
        X_cyrillic.append(img_array)
        y_cyrillic_text.append(row['Actual word'])
    if len(X_cyrillic) >= 1000:
        break

X_cyrillic = np.array(X_cyrillic)

print("Latin veri yüklendi:", X_latin.shape)
print("Kiril veri yüklendi:", X_cyrillic.shape)

# Latin etiketlerini temizle
y_latin_int = []
for label in y_latin:
    try:
        y_latin_int.append(int(label))
    except:
        y_latin_int.append(0)
y_latin_int = np.array(y_latin_int, dtype=np.int32)

# Kiril etiketlerini temizle (sadece ilk harf)
y_cyrillic_int = []
for text in y_cyrillic_text:
    if isinstance(text, str) and len(text) > 0:
        try:
            y_cyrillic_int.append(ord(text[0]))   # basit karakter kodu
        except:
            y_cyrillic_int.append(0)
    else:
        y_cyrillic_int.append(0)
y_cyrillic_int = np.array(y_cyrillic_int, dtype=np.int32)

# Tüm etiketleri birleştir ve yeniden numaralandır
all_labels = np.concatenate([y_latin_int, y_cyrillic_int])
unique_labels = np.unique(all_labels)
label_to_new_idx = {old: new for new, old in enumerate(unique_labels)}

y_latin_new = np.array([label_to_new_idx[label] for label in y_latin_int], dtype=np.int32)
y_cyrillic_new = np.array([label_to_new_idx[label] for label in y_cyrillic_int], dtype=np.int32)

print("Toplam benzersiz sınıf sayısı:", len(unique_labels))


-Bu kod, Latin ve Kiril etiketlerini temizliyor ve ortak bir sayı aralığına dönüştürüyor.

# Verileri birleştir
X_train = np.concatenate([X_latin, X_cyrillic], axis=0)
y_train = np.concatenate([y_latin_new, y_cyrillic_new], axis=0)

# Karıştır
indices = np.random.permutation(len(X_train))
X_train = X_train[indices]
y_train = y_train[indices]

# Train / Validation split
split = int(0.8 * len(X_train))
X_val = X_train[split:]
y_val = y_train[split:]
X_train = X_train[:split]
y_train = y_train[:split]

# Model
model = models.Sequential([
    layers.Conv2D(32, (3,3), activation='relu', input_shape=(28, 28, 1)),
    layers.MaxPooling2D((2,2)),
    layers.Conv2D(64, (3,3), activation='relu'),
    layers.MaxPooling2D((2,2)),
    layers.Flatten(),
    layers.Dense(128, activation='relu'),
    layers.Dropout(0.5),
    layers.Dense(len(unique_labels), activation='softmax')
])

model.compile(optimizer='adam', 
              loss='sparse_categorical_crossentropy', 
              metrics=['accuracy'])

model.summary()

# Eğitim
history = model.fit(
    X_train[..., np.newaxis],
    y_train,
    epochs=15,
    batch_size=64,
    validation_data=(X_val[..., np.newaxis], y_val),
    verbose=1
)

print("\nModel eğitimi tamamlandı.")

# Türkçe → Kiril test
test_words = ["Merhaba", "İstanbul", "Nasılsın", "Teşekkürler", "Türkiye", "Ankara", "Selam"]

def turkish_to_cyrillic(word):
    translit = {'a':'а','b':'б','c':'ц','ç':'ч','d':'д','e':'е','f':'ф','g':'г','h':'х','ı':'ы','i':'и','j':'ж','k':'к','l':'л','m':'м','n':'н','o':'о','ö':'ө','p':'п','r':'р','s':'с','ş':'ш','t':'т','u':'у','ü':'ү','v':'в','y':'й','z':'з'}
    return ''.join(translit.get(c.lower(), c) for c in word)

print("\nTürkçe Kelime → Kiril Test Sonuçları")
for word in test_words:
    cyr = turkish_to_cyrillic(word)
    print(f"{word:12} → {cyr}")


-Bu kod, modelin doğruluğunu gösterir ve Türkçe kelimelerin Kiril karşılıklarını test eder.
